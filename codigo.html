#include <Servo.h>

      Servo handServo;
      Servo boxServo;

      int switchStatus=0, action=0, vot, randNumber=0;
      const int ledPin = 13;
      const int frontSwitchPin = 2;
      const int handServoPin = 5;
      const int boxServoPin = 6;


      void setup()
      {
      Serial.begin(9600);

      pinMode(ledPin, OUTPUT);

      handServo.attach(handServoPin);
      boxServo.attach(boxServoPin);
      handServo.write(40.5);
      boxServo.write(40.5);

      //randomSeed(analogRead(0));
      }


      void loop()
      {
      switchStatus = digitalRead(frontSwitchPin);
      Serial.print("Switch");
      Serial.println(digitalRead(frontSwitchPin));
      //action = random(1,16);

      if (switchStatus == LOW){
      //Pruebas varias
      action = 0;
      }
      if (action == 0){
      Serial.println(boxServo.read());
      boxServo.write(40);
      delay(2000);
      Serial.println(boxServo.read());
      boxServo.write(0);
      delay(2000);
      boxServo.write(90);

      Serial.println(handServo.read());
      handServo.write(40);
      delay(2000);
      Serial.println(handServo.read());
      handServo.write(0);
      delay(2000);
      handServo.write(90);
      action ++;
      }

      else{

      boxServo.write(90);
      handServo.write(90);
      action ++;

      }
      }
